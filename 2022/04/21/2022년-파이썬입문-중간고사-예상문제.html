def f(arr): 
    arr2 = np.cumsum(arr)
    bmat = np.zeros([sum(arr) ,sum(arr) ])
    a,b=0,0 
    for i in arr2: 
        b=i 
        bmat[a:b,a:b] = np.ones(b-a) 
        a=i
    return bmat
